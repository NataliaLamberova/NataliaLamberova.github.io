get_results <- function(data){

table.9.variables.of.interest <- c("factor(rfeeinc)1", "factor(ta)1:factor(rfeeinc)1")
model.5 <- lm_robust(data = data[data$fee_inc_status == "Inc",], 
                     rating ~ factor(ta) + factor(rfeeinc) + factor(ta):factor(rfeeinc) 
                     + factor(master_id) + factor(month) + factor(chaincode) +
 # Firm-level controls
    otherfees + stars + price + wateronsite+ winteronsite + tennisonsite + golfonsite + casino + fitnessclasses + restaurants + concierge + turndown + indoorpools + outdoorpools + beachpoolstuff + factor(roomservcode) +  factor(spacode) + factor(fitnesscode) +
   # Traveler-level controls
                      factor(travtype)
                     , clusters = master_id , se_type = "stata")
coefs <- model.5$coefficients[names(model.5$coefficients) %in% table.9.variables.of.interest]
errors <- model.5$std.error[names(model.5$coefficients) %in% table.9.variables.of.interest]

results <- rbind(coefs = coefs, errors = errors)
}
